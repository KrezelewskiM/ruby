def remove(event)
  Pubnub.logger.debug('Pubnub') { "#{self.class}#remove" }
  kill_requester

  remove_channels(event)
  remove_groups(event)

  if @channel.empty? && @group.empty?
    finish
  else
    restart_heartbeat if @heart
  end
end
